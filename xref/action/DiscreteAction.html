90  	public int compareTo(DiscreteActionInterface c) {
91  		if (this.lapsTime == null) { // no lapstime is equivalent to infinity 
92  			return 1;
93  		}
94  		if (c.getCurrentLapsTime() == null) {// no lapstime is equivalent to infinity 
95  			return -1;
96  		}
97      	if(this.lapsTime > c.getCurrentLapsTime()){
98      		return 1;
99      	}
100     	if(this.lapsTime < c.getCurrentLapsTime()){
101     		return -1;
102     	}
103 		if(this.lapsTime == c.getCurrentLapsTime()){
104 			return 0;
105 		}
106 		return 0;
107 	}
