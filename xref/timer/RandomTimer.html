12  public class RandomTimer implements Timer {

14  	public enum randomDistribution {
15  		POISSON, EXP, POSIBILIST, GAUSSIAN;
16  	}
17  	
18  	private Random r = new Random();
19  	private randomDistribution distribution;
20  	private double rate;
21  	private double mean;
22  	private double lolim;
23  	private double hilim;
24  	
25  	
26  	public static randomDistribution string2Distribution(String distributionName){
27  		return Enum.valueOf(RandomTimer.randomDistribution.class, distributionName.toUpperCase());
28  	}	
29  	public static String distribution2String(randomDistribution distribution){
30  		return distribution.name();
31  	}

33  	/**
34  	 * @param param constraint 
35  	 * @throws Exception 
36  	 */
37  	public RandomTimer(randomDistribution distribution, double param) throws BadTimerException{
38  		if(distribution == randomDistribution.EXP ){
39  			this.distribution = distribution;
40  			this.rate = param;
41  			this.mean = 1/param;
42  			this.lolim = 0;
43  			this.hilim = Double.POSITIVE_INFINITY;
44  		}else if(distribution == randomDistribution.POISSON){
45  			this.distribution = distribution;
46  			this.rate = Double.NaN;
47  			this.mean = param;
48  			this.lolim = 0;
49  			this.hilim = Double.POSITIVE_INFINITY;
50  		}else{
51  			throw new BadTimerException("Bad Timer constructor for selected distribution");
52  		}
53  	}
54  	/**
55  	 * @param lolim
56  	 * @param hilim
57  	 * @throws Exception 
58  	 */
59  	public RandomTimer(randomDistribution distribution, int lolim, int hilim) throws BadTimerException{
60  		if(distribution == randomDistribution.POSIBILIST || distribution == randomDistribution.GAUSSIAN){
61  			this.distribution = distribution;
62  			this.mean = lolim + (double) (hilim - lolim)/2;
63  			this.rate = Double.NaN;
64  			this.lolim = lolim;
65  			this.hilim = hilim;
66  		}else{
67  			throw new BadTimerException("Bad Timer constructor for selected distribution");
68  		}
69  	}
70  	
71  	public String getDistribution(){
72  		return this.distribution.name();
73  	}
74  	
75  	public String getDistributionParam() {
76  		if(distribution == randomDistribution.EXP ){
77  			return "rate: " + this.rate;
78  		}else if(distribution == randomDistribution.POISSON){
79  			return "mean: " + this.mean;
80  		}else if(distribution == randomDistribution.POSIBILIST || distribution == randomDistribution.GAUSSIAN){
81  			return "lolim: " + this.lolim + " hilim: " + this.hilim;
82  		}
83  		
84  		return "null";
85  	}
86  	
87  	public double getMean(){
88  		return this.mean;
89  	}
90  	
91  	public String toString(){
92  		String s = this.getDistribution();
93  		switch (this.distribution){
94  		case POSIBILIST :
95  			s += " LoLim:" + this.lolim + " HiLim:" + this.hilim;
96  			break;
97  		case POISSON :
98  			s += " mean:" + this.mean;
99  			break;
100 		case EXP :
101 			s += " rate:" + this.rate;
102 			break;
103 		case GAUSSIAN :
104 			s += " LoLim:" + this.lolim + " HiLim:" + this.hilim;
105 			break;
106 		}
107 		
108 		return s;
109 	}

112 	/* (non-Javadoc)
113 	 * @see methodInvocator.Timer#next()
114 	 */
115 	@Override
116 	public Integer next(){
117 		switch (this.distribution){
118 		case POSIBILIST :
119 			return this.nextTimePosibilist();
120 		case POISSON :
121 			return this.nextTimePoisson();
122 		case EXP :
123 			return this.nextTimeExp();
124 		case GAUSSIAN :
125 			return this.nextTimeGaussian();
126 		}
127 		return -1; // Theoretically impossible !!!
128 	}
129 	
130 	/*
131 	 * Equivalent to methodInvocator.RandomTimer#next()
132 	 * 
133 	 * @param since has no effect
134 	 * 
135 	 * @see methodInvocator.RandomTimer#next(int)
136 	 */
137 
138 	
139 	/**
140 	 * Give good mean
141 	 * Give wrong variance  
142 	 */
143 	private int nextTimePosibilist(){
144 	    return (int)this.lolim + (int)(this.r.nextDouble() * (this.hilim - this.lolim));
145 	}
146 	
147 	/**
148 	 * Give good mean
149 	 * Give wrong variance  
150 	 */
151 	private int nextTimeExp(){
152 	    return (int)(-Math.log(1.0 - this.r.nextDouble()) / this.rate);
153 	}
154 	
155 	
156 	/**
157 	 * Give good mean
158 	 * Give good variance
159 	 */
160 	private int nextTimePoisson() {
161 	    
162 	    double exp = Math.exp(-this.mean);
163 	    int k = 0;
164 	    double p = 1.0;
165 	    do {
166 	        p = p * this.r.nextDouble();
167 	        k++;
168 	    } while (p > exp);
169 	    return k - 1;
170 	}   		
171 	    
172 	
173 	private int nextTimeGaussian(){
174 		return (int)this.lolim + (int)((this.r.nextGaussian() + 1.0)/2.0 * (this.hilim - this.lolim));
175 	}
176 	
177 	
178 	@Override
179 	public boolean hasNext() {
180 		return true;
181 	}
